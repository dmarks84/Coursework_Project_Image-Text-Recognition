The Assignment¶

Take a ZIP file) of images and process them, using a library built into python that you need to learn how to use. A ZIP file takes several different files and compresses them, thus saving space, into one single file. The files in the ZIP file we provide are newspaper images (like you saw in week 3). Your task is to write python code which allows one to search through the images looking for the occurrences of keywords and faces. E.g. if you search for "pizza" it will return a contact sheet of all of the faces which were located on the newspaper page which mentions "pizza". This will test your ability to learn a new (library), your ability to use OpenCV to detect faces, your ability to use tesseract to do optical character recognition, and your ability to use PIL to composite images together into contact sheets.

Each page of the newspapers is saved as a single PNG image in a file called images.zip. These newspapers are in english, and contain a variety of stories, advertisements and images. Note: This file is fairly large (~200 MB) and may take some time to work with, I would encourage you to use small_img.zip for testing.

# Find faces from array file for each file
def getfaces(img_pil,dic,img_array_g):
    images = []
    dic['faces'] = face_cascade.detectMultiScale(img_array_g,dic['scale'])
    for x,y,w,h in dic['faces']:
        images.append(img_pil.crop((x,y,x+w,y+h)))
        dic['images'].append(img_pil.crop((x,y,x+w,y+h)))
    return images

# Create contact sheet
def makecs(images):
    span = 5
    image_base = images[0]
    base_w = image_base.width
    base_h = image_base.height
    resize = (base_w,base_h)
    cs_width = base_w * span
    cs_rows = len(images) // span
    if len(images) % span > 0:
        cs_rows += 1
    cs_height = cs_rows * base_h
    w_inc = base_w
    h_inc = base_h
    cs = Image.new(image_base.mode, (cs_width,cs_height))
    x = 0
    y = 0
    for img in images:
        img.thumbnail(resize)
        img = img.resize(resize)
        cs.paste(img,(x, y))
        if x + w_inc == cs_width:
            x = 0
            y = y + h_inc
        else:
            x = x + w_inc
    ratio = 600 / cs_width
    cs = cs.resize((int(cs_width*ratio),int(cs_height*ratio)))
    return cs

project = {}

# Specify the correct zip file to read through
workingzip = 'readonly/small_img.zip'
# workingzip = 'readonly/images.zip'

# Search string defined
search = 'Christopher'
# search = 'Mark'

# Open zipfile
with ZipFile(workingzip,'r') as zipfile:
    files = zipfile.infolist()

# Define dictionaries within dictionary for each file name
    i = 0
    for file in files:
        project[file.filename] = {}
        project[file.filename]['images'] = []
        if workingzip == 'readonly/small_img.zip':
            l_scales = [2.2,2,2,1.85]
        elif workingzip == 'readonly/images.zip':
            l_scales = [2.2,1.7,2,2,2,1.7,2,1.85,2,2,2,2,5,2]
        project[file.filename]['scale'] = l_scales[i]
        i += 1

# Open zipfile
with ZipFile(workingzip,'r') as zip_file:
    files = zip_file.infolist()
    print("Zip folder under consideration:",workingzip)
    print("String searched:",search)
    printstring = 'Files found: '
    for file in files:
        printstring += file.filename
        printstring += '; '
    printstring += '\n'
    print(printstring)
    
    # Look at each file to create image arrays
    for file in files:
        print("Analyzing file",file.filename)

        with zip_file.open(file) as img_file:

            img_pil = Image.open(img_file)
            img_pil_g = img_pil.convert('L')
            img_pil_g.save('tempfile.png')
            img_array=cv.imread('tempfile.png')
            img_array_g = cv.cvtColor(img_array, cv.COLOR_BGR2GRAY)

            # Perform word search
            print("Searching for '{}' in file".format(search),file.filename)
            g_text = pytesseract.image_to_string(img_pil_g) 
            g_words = g_text.strip().split()
            found = False
            for word in g_words:
                if word.find(search) == 0:
                    found = True
                    break
            if found == True:
                print("Results were matched in this file for the searched string, '{}'".format(search))
                images = getfaces(img_pil,project[file.filename],img_array_g)
                if len(images) == 0:
                    print("But, there were no faces found in the file\n")
                else:
                    display(makecs(images))
            else:
                print("No matches found for that text\n")
            img_pil.close()
